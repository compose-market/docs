---
title: "Get Agent Metadata"
api: "GET https://manowar.compose.market/agent/{id}"
description: "Get metadata for a registered agent."
---

Returns metadata for an agent registered in the Manowar runtime.

## Path Parameters

<ParamField path="id" type="string" required>
  Agent identifier: wallet address or agent ID
</ParamField>

## Response

<ResponseField name="name" type="string">
  Agent display name
</ResponseField>

<ResponseField name="description" type="string">
  Agent description
</ResponseField>

<ResponseField name="walletAddress" type="string">
  Agent wallet address
</ResponseField>

<ResponseField name="dnaHash" type="string">
  On-chain DNA hash
</ResponseField>

<ResponseField name="agentCardUri" type="string">
  IPFS URI to agent metadata
</ResponseField>

<ResponseField name="creator" type="string">
  Creator wallet address
</ResponseField>

<ResponseField name="model" type="string">
  LLM model ID
</ResponseField>

<ResponseField name="plugins" type="array">
  Available tool plugins
</ResponseField>

<ResponseField name="endpoints" type="object">
  API endpoints:
  - `chat` - Chat endpoint
  - `stream` - Streaming endpoint
</ResponseField>
